\section{Context Sensitivity}

\begin{frame}{The Wild West of Grammars}
    \textbf{Context-Sensitive Grammars (CSGs)} are powerful:

    \textbf{General CSG production form:}
    \[
        x S y \to x \Phi y, \quad \Phi \ne \epsilon
    \]

    Non-terminal $S$ rewrites to $\Phi$ in context $x \cdots y$

    \vspace{3mm}

    \textbf{Our focus:} Attribute Grammars (a specific type of CSG)
\end{frame}

\begin{frame}{Attribute Grammars}
    \textbf{Attribute Grammar (AG):} CSG where each symbol has associated attributes

    \vspace{2mm}

    \textbf{Intuition:}
    \begin{itemize}
        \item Start with a CFG
        \item Add rules for computing attribute values
        \item Values depend on context
    \end{itemize}

    \vspace{2mm}

    \textbf{Common uses:}
    \begin{itemize}
        \item Semantic analysis in compilers
        \item Type checking
        \item Code generation
    \end{itemize}
\end{frame}

\begin{frame}{Morphophonemic Alternation is regular but AGs are a better represent it}
    Recall morphophonemic alternation from~\cref{sec:particle_err}.

    \vspace{2mm}

    \textbf{Instead of} feature tagging (which explodes),\\
    \textbf{use} an attribute grammar to handle context sensitivity:


    \begin{align*}
        S               & \to \Phi \; \text{raw} & : \Phi \in \mathbb{NOUN} \cup \mathbb{VERB} \cup \mathbb{ADJ} \cup \mathbb{ADV}, \; \operatorname{last}(\Phi) \in \{\text{vowels}\}    \\
                        & \quad \mid \Phi \; \text{daw} & : \Phi \in \mathbb{NOUN} \cup \mathbb{VERB} \cup \mathbb{ADJ} \cup \mathbb{ADV}, \; \operatorname{last}(\Phi) \notin \{\text{vowels}\}
    \end{align*}
    where $\operatorname{last}(\Phi) = \Phi_{|\Phi| - 1}$.

    \textbf{Key:} The condition checks context (last character) without exploding the grammar.
\end{frame}

\begin{frame}
    \textbf{Benefits of Attribute Grammars:}
    \begin{itemize}
        \item \textbf{Compactness:} Avoids combinatorial explosion of productions
        \item \textbf{Maintainability:} Easier to update rules without rewriting large parts
        \item \textbf{Clarity:} Clearly separates structure from context-dependent rules
    \end{itemize}

    \vspace{3mm}

    \textbf{Trade-off:} AGs in \textbf{PSPACE-complete}!
\end{frame}
