\section{Formal Languages}

\begin{frame}{Formal Languages + Chomsky}
    \textbf{Formal Language $\mathcal{L}$} over alphabet $\Sigma$:
    \begin{itemize}
        \item Set of all \textbf{valid} strings from $\Sigma$~\footnote{E.g., ``kamusta'' $\in \mathcal{F}$ but ``asdfgh'' $\notin \mathcal{F}$; thus $\mathcal{F}\subseteq \Sigma^*$}
    \end{itemize}

    \vspace{5mm}

    \textbf{Chomsky Hierarchy} classifies languages into four levels:
    \begin{itemize}
        \item \textbf{Regular} --- Regular grammars, finite automata
        \item \textbf{Context-Free} --- CFGs, pushdown automata
        \item \textbf{Context-Sensitive} --- CSGs, linear-bounded automata
        \item \textbf{Recursively Enumerable} --- Unrestricted grammars, Turing machines
    \end{itemize}
\end{frame}
