\subsection{Reduplication is midly context-sensitive}
\begin{frame}{Habitual Action in Tagalog is Midly Context-Sensitive}
    \textbf{Habitual Action} denotes that a verb is done regularly.

    \vspace{3mm}

    \textbf{Rule:} enclose the word ``nang'' with a reduplication of the verb.

    \textbf{Examples:}
    \begin{enumerate}
        \item Kain \textbf{nang} kain
        \item Tulog \textbf{nang} tulog
    \end{enumerate}
\end{frame}

\begin{frame}{Habitual Action is at least Regular}
    All finite (a union of singletons) languages $L$ are regular (think of this as brute forcing all singletons as a production from the start symbol).

    \vspace{5mm}

    So, since the set of verbs $\mathbb{VERB} = \{v_0, v_1, \ldots, v_n\}$ is
    finite, the habitual action language is also regular.
\end{frame}

\begin{frame}{Habitual Action is Not Practical as Regular}
    But again, this is not a practical grammar! It requires enumerating all verbs in the language, which is finite but very massive!

    \vspace{5mm}

    Would implementing a CFG be any better?
\end{frame}

\begin{frame}{Habitual Action is \emph{Still} Not Practical as Context-Free}
    Since all finite languages are regular, they are also context free ($\textbf{REG}\subseteq\textbf{CFG}$).

    \vspace{5mm}

    But this still requires enumerating all verbs in the language, which is finite
    but very massive!
    \[
        S \to v_0 \; \text{nang} \; v_0 \;\mid\; v_1 \; \text{nang} \; v_1 \;\mid\; \ldots \;\mid\; v_n \; \text{nang} \; v_n \; \mid
    \]
    \vspace{5mm}

\end{frame}

\begin{frame}{Habitual Action = Copy Language}
    If we consider the $\mathbb{VERB}$ set to be infinite, the language $\textbf{HA} = \left\{v \; \text{nang} \; v \mid v \in \mathbb{VERB}\right\}$
    can be seen as analogous to the $\textbf{Copy-Language} = \left\{ww \mid w \in \Sigma^*\right\}$. From this, it follows that the habitual action construction
    is context-sensitive.
\end{frame}

\begin{frame}{Habitual Action is Midly Context-Sensitive}
    Mildly Context-Sensitive Languages (\textbf{MCSLs}) are a subclass of context-sensitive languages, they occupy the boundary between \textbf{CFL} and \textbf{CSL}s,\footnote{They can be thought of as the simplest or least complex members of the set of \textbf{CSL}---complex enough to exceed the expressive power of \textbf{CFLs}, yet not as \emph{hard} as the more general \textbf{CSLs}.}. 

    \begin{align*}
        \mathbf{S} & \to X\ \text{``nang''}\ Y
                   &                                                       & \{ \; X.\texttt{word}=Y.\texttt{word} \;\} \\
        X          & \to \text{``kain''} \mid \text{``tulog''} \mid \ldots                                              \\
        Y          & \to \text{``kain''} \mid \text{``tulog''} \mid \ldots
    \end{align*}

    \textbf{Remark.} You could remove the $Y$ non-terminal and just have $X$ generate the verb again, but this is to illustrate the copying constraint more clearly.
\end{frame}
